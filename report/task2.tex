\begin{equation}
	\begin{cases}
		$$
		\rho_0 (x) = 1, \ \ x < 4.5 \ \text{или} \  x > 5.5,
		\\
		\rho_0 (x) = 2, \ \ x \in [4.5; 5.5], 
		\\
		u_0 (x) \equiv 0, \ \ x \in [0; 10], 
		\\
		u (t, 0) = u (t, 10) = 0, \ \ t \in [0; T], 
		\\
		f \equiv 0.
		$$
	\end{cases}
\end{equation}

\begin{equation}
	\begin{cases}
		$$
		u_0 (x) = 0, \ \ x < 4.5 \ \text{или} \  x > 5.5,
		\\
		u_0 (x) = 1, \ \ x \in [4.5; 5.5], 
		\\
		\rho_0 (x) \equiv 1, \ \ x \in [0; 10], 
		\\
		u (t, 0) = u (t, 10) = 0, \ \ t \in [0; T],
